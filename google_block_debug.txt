function initGoogleSignIn() {

      alert("Google no devolvió credential");
      return;
    }

      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ id_token: response.credential })
    });

    if (!data || !data.ok || !data.token) {
      console.error("AUTH_FAIL:", data);
      alert("Error al ingresar con Google");
      return;

  overlay.innerHTML = `
    <div style="max-width:360px;width:100%;text-align:center;">
      <div style="font-size:2rem;margin-bottom:8px;">👋</div>
      <h2 style="color:#00E5FF;font-family:Rajdhani,sans-serif;font-size:1.4rem;margin-bottom:6px;">Hola, ${nombre}</h2>
      <p style="color:#64748b;font-size:0.85rem;margin-bottom:28px;">Bienvenido a ServiRed. ¿Qué querés hacer?</p>
      <div style="display:flex;flex-direction:column;gap:12px;">
        <button onclick="elegirRol(this,'CLIENTE')" style="background:#111827;border:1px solid #00E5FF;color:#00E5FF;padding:14px 20px;border-radius:10px;font-size:0.95rem;font-family:Rajdhani,sans-serif;cursor:pointer;text-align:left;">
          🔍 Buscar un trabajador o servicio
        </button>
        <button onclick="elegirRol(this,'TRABAJADOR')" style="background:#111827;border:1px solid #FF6D00;color:#FF6D00;padding:14px 20px;border-radius:10px;font-size:0.95rem;font-family:Rajdhani,sans-serif;cursor:pointer;text-align:left;">
          🔧 Ofrecer mis servicios
        </button>
        <button onclick="elegirRol(this,'COMERCIO')" style="background:#111827;border:1px solid #39ff14;color:#39ff14;padding:14px 20px;border-radius:10px;font-size:0.95rem;font-family:Rajdhani,sans-serif;cursor:pointer;text-align:left;">
          🏪 Registrar mi comercio
        </button>
      </div>
    </div>
  `;
  document.body.appendChild(overlay);
}

function elegirRol(btn, rol) {
  document.querySelectorAll("#rol-selector-overlay button").forEach(b => b.disabled = true);
  btn.textContent = "Cargando...";
  if (rol === "CLIENTE") { location.reload(); }
  else if (rol === "TRABAJADOR") { iniciarWizardProvider(document.getElementById("rol-selector-overlay")); }
  else if (rol === "COMERCIO") { document.getElementById("rol-selector-overlay").remove(); abrirModal("onboarding-comercio"); }
}

window.addEventListener("load", function() {
  if (window.google && window.google.accounts) { initGoogleSignIn(); }
  else { document.addEventListener("googleAccountsReady", initGoogleSignIn); }
  setTimeout(initGoogleSignIn, 2000);
  setTimeout(renderHomeGoogleBtn, 2500);
});

function renderHomeGoogleBtn() {
  if (!window.google || !window.google.accounts) { setTimeout(renderHomeGoogleBtn, 500); return; }
  var el = document.getElementById("home-google-btn");
  if (!el) return;
  google.accounts.id.renderButton(el, { theme:"filled_black", size:"large", text:"continue_with", locale:"es_419", width:320 });
}
